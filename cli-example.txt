// Setup CLI

>>> conda activate tcdb-env
>>> pip install -e .


// Matplotlib Backend Adjustment

>>> export MPLBACKEND=Agg


/* 
Example 1 - Getting Training Data Through CDD File
 -c: CDD mode, takes in 1 cdd file to parse
 -d: processed data store location, in default folder output/
 --log-level ERROR: avoid verbose 
*/

>>> tcdb-visualize -c data/cdd_files/singles.cdd -d data.csv --log-level ERROR


/*
Example 2 - Getting Training Data Through Resc Files
 -r: Rescue mode, takes in rescue file folder
 -d: processed data store location, in default folder output/
 --log-level ERROR: avoid verbose
*/

>>> tcdb-visualize -r data/rescue_files/ -d data.csv --log-level ERROR

/*
Example 3 - Plotting using CDD File
 -t: TCIDs (family) to process
 -c: CDD mode, takes in 1 cdd file to parse
 -p: Takes in plotting options (all|[general,char,arch,summary])
 --log-level ERROR: avoid verbose
*/

>>> tcdb-visualize -t 1.A.102 -c data/cdd_files/singles.cdd -p all --log-level ERROR
